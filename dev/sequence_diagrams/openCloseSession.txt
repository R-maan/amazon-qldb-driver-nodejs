title Session Open and Close
actor application

application->*QldbDriver:<<create>>(ledgerName)
participant QldbSession
QldbDriver-->>application:QldbDriver
application->QldbDriver:getSession()
QldbDriver->*Communicator:Communicator.create(ledgerName)
database QLDB
Communicator->QLDB:StartSessionRequest
QLDB-->>Communicator:StartSessionResponse
Communicator-->>QldbDriver:Communicator
QldbDriver->*QldbSession:<<create>>(Communicator)
QldbSession-->>QldbDriver:QldbSession
QldbDriver-->>application:QldbSession
application->QldbSession:close()
QldbSession->Communicator:endSession()
Communicator->QLDB:EndSessionRequest
QLDB-->>Communicator:EndSessionResponse
Communicator-->>QldbSession:
QldbSession-->>application:
application->QldbDriver:close()
QldbDriver-->>application:
